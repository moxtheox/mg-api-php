/**
 * Executes multiple callables concurrently and suspends the caller 
 * until all are completed.
 * * @param array<string, callable> $tasks
 * @return array<string, mixed> The results keyed by the original task keys.
 */
public function waitAll(array $tasks): array {
    $parentFiber = Fiber::getCurrent() ?? throw new RuntimeException("waitAll must run in a Fiber.");
    $results = [];
    $remaining = count($tasks);

    foreach ($tasks as $key => $task) {
        $this->register(function() use ($task, $key, &$results, &$remaining, $parentFiber) {
            try {
                $results[$key] = $task();
            } catch (\Throwable $e) {
                $results[$key] = $e; // Capture exceptions to prevent one failure from killing the set
            } finally {
                $remaining--;
                if ($remaining === 0 && $parentFiber->isSuspended()) {
                    $parentFiber->resume($results);
                }
            }
        });
    }

    // Suspend the main script fiber until the last child finishes
    return Fiber::suspend();
}

public function getHandle(string $url): \CurlHandle {
    $host = parse_url($url, PHP_URL_HOST) ?: 'default';
    if (!empty($this->handlePool[$host])) {
        return array_pop($this->handlePool[$host]);
    }
    return curl_init();
}

public function releaseHandle(\CurlHandle $ch): void {
    $info = curl_getinfo($ch);
    $host = parse_url($info['url'] ?? '', PHP_URL_HOST) ?: 'default';
    
    curl_reset($ch);
    $this->handlePool[$host][] = $ch;
}
3. The waitAll Abstract Pattern
Now that the plumbing is fixed, here is the abstracted waitAll in Reactor.php that lets you work with the data after the returns:

PHP
/**
 * Executes a batch of tasks and returns an associative array of results.
 */
public function waitAll(array $tasks): array {
    $parent = \Fiber::getCurrent() ?? throw new \RuntimeException("waitAll requires a Fiber.");
    $results = [];
    $count = count($tasks);

    foreach ($tasks as $key => $task) {
        $this->register(function() use ($task, $key, &$results, &$count, $parent) {
            try {
                $results[$key] = $task();
            } catch (\Throwable $e) {
                $results[$key] = $e;
            } finally {
                if (--$count === 0) $parent->resume();
            }
        });
    }

    \Fiber::suspend();
    return $results;
}
Why this works:
Persistence: By setting CURLOPT_URL on a borrowed handle, you keep the SSL connection alive but point it at the correct endpoint.

State Machine: The "nudge" in await ensures that $active in the run() loop won't stay at 0 if you only have one pending request.

Concurrency: In index.php, you can now do $data = $reactor->waitAll([...]) and foreach ($data['users']->data ...) safely.
