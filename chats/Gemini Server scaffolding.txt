/**
 * LiveStreamer: Manages the connection to the Bun/PHP feed
 */
class LiveStreamer {
    constructor(url) {
        this.url = url;
        this.charBuffer = []; // Store characters for the rain
        this.socket = null;
        this.statusEl = document.getElementById('status');
        this.init();
    }

    init() {
        this.socket = new WebSocket(this.url);
        
        this.socket.onopen = () => {
            this.statusEl.innerText = "CONNECTED: PHP Feed Active";
            this.statusEl.style.color = "#00ff41";
        };

        this.socket.onmessage = (event) => {
            // Bun is sending strings (lines from PHP stdout)
            const text = event.data;
            // Push each character into our rain buffer
            for (let i = 0; i < text.length; i++) {
                this.charBuffer.push(text[i]);
            }

            // Optional: Prevent memory leaks if PHP sends data faster than the rain falls
            if (this.charBuffer.length > 10000) {
                this.charBuffer.splice(0, 1000); 
            }
        };

        this.socket.onclose = () => {
            this.statusEl.innerText = "DISCONNECTED: Retrying...";
            this.statusEl.style.color = "#ff0000";
            setTimeout(() => this.init(), 2000);
        };
    }

    getNextChar() {
        // Return next char from PHP stream, or a random hex digit if buffer is empty
        if (this.charBuffer.length > 0) {
            return this.charBuffer.shift();
        }
        return Math.floor(Math.random() * 16).toString(16); 
    }
}

const canvas = document.getElementById('canvas');
const ctx = canvas.getContext('2d');

// Match the endpoint in server.ts
const streamer = new LiveStreamer('ws://localhost:3000/geotab-feed');

const fontSize = 18;

// RainColumn class remains largely the same, but uses the new streamer
class RainColumn {
    constructor(x, streamer, fontSize) {
        this.x = x;
        this.fontSize = fontSize;
        this.streamer = streamer;
        this.reset();
    }

    reset() {
        this.y = Math.random() * -1000;
        this.speed = (Math.random() * 2) + 1; 
        this.lastUpdate = 0;
    }

    draw(ctx, timestamp, height) {
        // Fall speed logic
        if (timestamp - this.lastUpdate > 50 / this.speed) {
            const char = this.streamer.getNextChar();
            
            // Draw the leading character (Bright White)
            ctx.fillStyle = '#fff';
            ctx.fillText(char, this.x, this.y);

            this.y += this.fontSize;
            this.lastUpdate = timestamp;

            if (this.y > height) {
                this.reset();
                this.y = 0;
            }
        }
    }
}

function resize() {
    canvas.width = window.innerWidth;
    canvas.height = window.innerHeight;
    const colCount = Math.floor(canvas.width / fontSize);
    columns = [];
    for (let i = 0; i < colCount; i++) {
        columns.push(new RainColumn(i * fontSize, streamer, fontSize));
    }
}

function animate(timestamp) {
    ctx.fillStyle = 'rgba(0, 0, 0, 0.15)';
    ctx.fillRect(0, 0, canvas.width, canvas.height);

    ctx.font = `bold ${fontSize}px monospace`;
    ctx.fillStyle = '#00ff41';

    columns.forEach(col => col.draw(ctx, timestamp, canvas.height));
    requestAnimationFrame(animate);
}

import { Elysia } from 'elysia'
import { staticPlugin } from "@elysiajs/static";

const app = new Elysia()
    // Serve the public folder so index.html is accessible at http://localhost:3000
    .use(staticPlugin({
        assets: 'public',
        prefix: '/'
    }))
    .ws('/geotab-feed', {
        async open(ws) {
            console.log("WebSocket opened. Starting PHP Feed...");

            // Ensure path matches the volume mount in compose.yaml (/usr/src/app)
            const proc = Bun.spawn(["php", "/usr/src/app/src/index.php"], {
                stdout: "pipe",
                stderr: "inherit",
            });

            (ws.data as any).proc = proc;

            (async () => {
                const reader = proc.stdout
                    .pipeThrough(new TextDecoderStream())
                    .getReader();

                let buffer = "";
                try {
                    while (true) {
                        const { value, done } = await reader.read();
                        if (done) break;

                        buffer += value;
                        const lines = buffer.split("\n");
                        buffer = lines.pop() || "";

                        for (const line of lines) {
                            if (line.trim()) {
                                // Send raw text to the browser
                                ws.send(line); 
                            }
                        }
                    }
                } catch (e) {
                    console.error("Stream error:", e);
                }
            })();
        },
        close(ws) {
            const proc = (ws.data as any).proc;
            if (proc) proc.kill("SIGINT"); 
        }
    })
    .listen(3000);

// Inside your <script> tag in index.html

class DigitalRainStreamer {
    constructor(url) {
        this.url = url;
        this.buffer = [];
        this.socket = null;
        this.statusEl = document.getElementById('status');
        this.init();
    }

    init() {
        // Correct the URL to match the Elysia route
        this.socket = new WebSocket(this.url);
        
        this.socket.onopen = () => {
            this.statusEl.innerText = "LIVE: PHP FEED CONNECTED";
        };

        this.socket.onmessage = (event) => {
            // Since the server sends ws.send(line), event.data is a string
            const text = event.data;
            for (let char of text) {
                this.buffer.push(char);
            }
            // Keep buffer from exploding during high-traffic competition feeds
            if (this.buffer.length > 5000) this.buffer.shift();
        };

        this.socket.onclose = () => {
            this.statusEl.innerText = "RECONNECTING...";
            setTimeout(() => this.init(), 2000);
        };
    }

    getNextChar() {
        // If buffer is empty, show a '·' or space to keep the rain moving
        return this.buffer.shift() || ' ';
    }
}
